SUBIR UN PROYECTO AL REPOSITORIO	git init
					git add .
					git commit -m "Nombre Proyecto"
					git branch -M main
					git remote add origin https://github.com/luis-pixel89/web-equipo.git
					git push -u origin main

CAMBIO y CREO RAMA		git checkout -b dev
CAMBIO DE RAMA			git checkout dev
VER RAMA			git branch
PUSH NUEVA RAMA			git push origin dev
CAMBIARSE DE RAMA CREADA	git fetch origin dev:dev
COMANDO DE RESTREO DE CAMBIOS	git fetch origin


MENSAJES DE COMMIT		feat: nueva funcionalidad
				fix: corrección de error
				style: cambios de diseño
				docs: actualización de documentación
